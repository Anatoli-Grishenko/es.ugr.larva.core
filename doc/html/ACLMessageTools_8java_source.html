99  public static String fancyWriteACLM(ACLMessage original, boolean simple) {
100  String res = "", sep = "|";
101  ACLMessage msg = (ACLMessage) original.clone();
102  res += (msg.getSender() == null ? _NULLVAL : "||SND" + sep + msg.getSender().getLocalName());
103  Iterator it;
104  it = msg.getAllReceiver();
105  res += "||RCV" + sep;
106  while (it.hasNext()) {
107  res += ((AID) it.next()).getLocalName() + " ";
108  }
109  res = res + "||CNT" + sep + (isJsonACLM(msg) ? trimString(msg.getContent(), 255) : msg.getContent()) + "||";
110  if (!simple) {
111  it = msg.getAllReplyTo();
112  res = "||PFM" + sep + ACLMessage.getPerformative(msg.getPerformative()) + res;
113  res += "RPT" + sep;
114  while (it.hasNext()) {
115  res += ((AID) it.next()).getLocalName() + " ";
116  }
117  res += (msg.getProtocol() == null ? _NULLVAL : "||PRT" + sep + msg.getProtocol())
118  + (msg.getConversationId() == null ? _NULLVAL : "||CNV" + sep + msg.getConversationId())
119  + (msg.getEncoding() == null ? _NULLVAL : "||ENC" + sep + msg.getEncoding())
120  + (msg.getReplyWith() == null ? _NULLVAL : "||RPW" + sep + msg.getReplyWith())
121  + (msg.getInReplyTo() == null ? _NULLVAL : "||IRT" + sep + msg.getInReplyTo())
122  + (msg.getLanguage() == null ? _NULLVAL : "||LAN" + sep + trimString(msg.getLanguage(), 10))
123  + (msg.getOntology() == null ? _NULLVAL : "||ONT" + sep + msg.getOntology());
124  res += "||";
125  }
126  return res;
