113  public Point placeAtMap(String where, ArrayList<Double> pos) {
114  Point res = new Point(0, 0, 0);
115  int width = this.getEnvironment().getSurface().getWidth(),
116  height = this.getEnvironment().getSurface().getHeight();
117  switch (where) {
118  case "choice":
119  res.setX(pos.get(0));
120  res.setY(pos.get(1));
121  break;
122  case "middle":
123  res.setX(width / 2);
124  res.setY(height / 2);
125  break;
126  case "N":
127  res.setX(width / 2);
128  res.setY(0);
129  break;
130  case "S":
131  res.setX(width / 2);
132  res.setY(height);
133  break;
134  case "E":
135  res.setX(width - 1);
136  res.setY(height / 2);
137  break;
138  case "W":
139  res.setX(0);
140  res.setY(height / 2);
141  break;
142  case "NE":
143  res.setX(width - 1);
144  res.setY(0);
145  break;
146  case "SE":
147  res.setX(width - 1);
148  res.setY(height - 1);
149  break;
150  case "SW":
151  res.setX(0);
152  res.setY(height - 1);
153  break;
154  case "NW":
155  res.setX(0);
156  res.setY(0);
157  break;
158  default:
159  do {
160  res.setX((int) (Math.random() * width));
161  res.setY((int) (Math.random() * height));
162  } while (this.getEnvironment().getSurface().getStepLevel(res.getX(), res.getY()) > this.maxflight);
163  }
164  if (0 <= res.getX() && res.getX() < width && 0 <= res.getY() && res.getY() < height) {
165  res.setZ(this.getEnvironment().getSurface().getStepLevel(res.getX(), res.getY()));
166  return res;
167  } else {
168  return new Point(0, 0, this.getEnvironment().getSurface().getStepLevel(0, 0));
169  }

340  public ArrayList<Thing> getDetectableList(Perceptor p) {
341  PROPERTY property = p.getProperty();
342  String type = p.getType();
343  Thing who = p.getOwner();
344  ArrayList<Thing> detectable = new ArrayList<>();
345 
346  // Intern sensors only perceive oneself
347  if (p.getSelection() == SELECTION.INTERN) {
348  detectable.add(who);
349  return detectable;
350  }
351  // An Thing is detectable if the propoerty of the sensor is
352  // visible on it and is the detected type
353  for (Thing t : _visibility.get(property)) {
354  if (!t.equals(who) && _ontology.matchTypes(t.getType(), type)) {
355  detectable.add(t);
356  }
357  }
358  // Sometimes only the closest Thing is detected
359  if (p.getSelection() == SELECTION.CLOSEST && detectable.size() > 1) {
360  Point mypos = p.getOwner().getPosition(), yourpos;
361  double shortest = Double.MAX_VALUE, distance;
362  Thing best = null, ti;
363  for (int i = 0; i < detectable.size(); i++) {
364  ti = detectable.get(i);
365  yourpos = ti.getPosition();
366  distance = mypos.fastDistanceXYTo(yourpos);
367  if (distance < shortest) {
368  shortest = distance;
369  best = ti;
370  }
371  }
372  detectable.clear();
373  detectable.add(best);
374  }
375  return detectable;
376  }
