257  public void addTrail(String ID, int x, int y, int z) {

262  trails.put(ID, new ATC_Trail(ID, Color.RED)); //colors[trails.keySet().size()]));
263  } else if (ID.startsWith("D")) {
264  trails.put(ID, new ATC_Trail(ID, Color.WHITE)); //colors[trails.keySet().size()]));
265  } else if (ID.startsWith("R")) {
266  trails.put(ID, new ATC_Trail(ID, Color.CYAN)); //colors[trails.keySet().size()]));

268  trails.put(ID, new ATC_Trail(ID, Color.YELLOW)); //colors[trails.keySet().size()]));
269  }
270  }

272  trails.get(ID).pushTrail(p);
