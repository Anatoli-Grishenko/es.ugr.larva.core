126  protected Point calculate3D(Point p3d) {
127  Point res = new Point(0, 0);
128  double xa, ya, za, xa2, ya2, za2, x2d, y2d;
129 
130  xa2 = p3d.getY();
131  ya2 = p3d.getX();
132  za2 = p3d.getZ();
133  xa2 = (-1) * xa2;
134  xa = cr1 * xa2 - sr1 * za2;
135  za = sr1 * xa2 + cr1 * za2;
136  xa2 = cr2 * xa + sr2 * ya2;
137  ya = cr2 * ya2 - sr2 * xa;
138  za2 = cr3 * za - sr3 * ya;
139  ya2 = sr3 * za + cr3 * ya;
140  xa2 = xa2 + mx;
141  ya2 = ya2 + my;
142  za2 = za2 + mz;
143  x2d = d * xa2 / za2;
144  x2d += this.getSize().width / 2;
145 // x2d += 399;
146  y2d = d * ya2 / za2;
147 // y2d += 299;
148  y2d += this.getSize().height / 2;
149  xa = pantalla_x / this.getSize().width;
150  ya = pantalla_y / this.getSize().height;
151  x2d *= xa;
152  y2d *= ya;
153  res = new Point(x2d, y2d);
154  posx = 0;
155  posy = 0;
156  return res;
157  }
